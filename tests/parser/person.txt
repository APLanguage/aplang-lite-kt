fn main() {
   var person = Person()
   println(person.toString())
   println("Before:", person.name)
   person.name = "AP"
   println("After:", person.name)
   println(person.sayHi("everyone"))
}

class Person {
  var name: String = "Default"

  fn sayHi(target: I64) : String {
    return name + ": Hi, " + target + "!"
  }

  fn toString() : String {
    return "Person(" + name + ")"
  }
}
